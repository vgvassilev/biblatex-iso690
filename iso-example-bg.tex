\documentclass[a4paper,10pt]{ltxdockit}
\usepackage[utf8]{inputenc}
\usepackage[T2A]{fontenc} % This has to go after the utf8 encoding and before babel
\usepackage[english, bulgarian]{babel} % Using bg in babel enables our ISO 690:2011

% Implements БДС ISO 690:2011.
\usepackage[
   backend=biber
  ,style=iso-numeric
  ,sorting=none % order by occurence
  ,giveninits=true
  ,autolang=other
  ,sortlocale=bg_BG
  ,bibencoding=UTF8
  %,block=ragged
]{biblatex}

\bibliography{iso-example-bg}
% Print author names as small caps
\let\lastnameformat=\textsc

% Print the bibliography list as [1], [2]...
\DeclareFieldFormat{labelnumberwidth}{[#1]}
\DeclareFieldFormat{shorthandwidth}{[#1]}
